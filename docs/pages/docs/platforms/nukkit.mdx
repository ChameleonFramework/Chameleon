### Maven
```xml filename="pom.xml"
<properties>
   <chameleon.version>VERSION</chameleon.version>
</properties>

<repositories>
    <!-- Remove this if you aren't using a snapshot version -->
    <repository>
       <id>sonatype-snapshots-s01</id>
       <url>https://s01.oss.sonatype.org/content/repositories/snapshots/</url>
    </repository>
</repositories>

<dependencies>
    <dependency>
        <groupId>dev.hypera</groupId>
        <artifactId>chameleon-platform-nukkit</artifactId>
        <version>${chameleon.version}</version>
    </dependency>
</dependencies>
```

### Gradle <small>(Kotlin)</small>
```kotlin filename="build.gradle.kts"
repositories {
    mavenCentral() // Release
    maven("https://s01.oss.sonatype.org/content/repositories/snapshots/") // Snapshot
}

dependencies {
    val chameleonVersion = "<version>"
    implementation("dev.hypera:chameleon-platform-nukkit:$chameleonVersion")
}
```
